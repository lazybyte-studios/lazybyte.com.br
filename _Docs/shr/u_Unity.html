Adicionando Sons

  • Criar a pasta "Resources" dentro da pasta de Assets
  • Baixar os áudios e colocar na pasta Resources
  • Criar um componente empty
  • Criar uma classe para controlar o áudio 

public class GerenciaSom : MonoBehaviour
{
    public static AudioClip audio01, audio02;
    static AudioSource fonteSom;

    void Start()
    {
        som1 = Resources.Load("nomeDoAudio");
        som2 = Resources.Load("nomeDoAudio2");

        fonteSom.GetComponent();
    }

    public static TocaSom( string audio )
    {
        switch( audio )
        {
            case "som1":
                fonteSom.PlayOneShot( som1 );
                break;
            case "som2":
                fonteSom.PlayOneShot( som2 );
                break;
        }
    }

}
